ACTOR DualMP40: VCWeapon
{
	Weapon.AmmoUse1 0 
	Weapon.SelectionOrder 5100
	Weapon.AmmoGive1 0 //How much ammo can be gathered with this weapon
	Weapon.AmmoUse2 0
	Weapon.AmmoGive2 0
	Weapon.AmmoType2 "P.Bullets" //Reserve ammo type
	Weapon.AmmoType1 "DoubleMP40Ammo" //Magazine ammo type
	Weapon.SisterWeapon "MP40"
	Obituary "%o was drilled by %k's Double Rifles." //Kill message
    AttackSound "None" //You better leave this clear
	Inventory.Pickupmessage "" //Pickup message
    +WEAPON.NOAUTOAIM //Makes this weapon never autoaim, so recoil can work better
    Scale 0.6 //Weapon's pickup sprite size
	+WEAPON.NOAUTOFIRE
	+WEAPON.NO_AUTO_SWITCH
	States
	{

	Spawn:
	TNT1 A 100//Pickup sprite
	Loop

	Ready:
	    TNT1 A 0
		TNT1 A 0 A_PlaySound("MP40UP", 1)
		MPDS AB 1
		TNT1 A 0 A_TakeInventory("StopDualWield", 1)
		TNT1 A 0 A_PlaySound("MP40UP", 3)
		MPDS CDEF 1
		TNT1 A 0 A_GunFlash
		Goto ReadyToFire

	RemoveDual:
	     TNT1 A 0 A_GiveInventory("MP40", 1)
		 TNT1 A 0 A_TakeInventory("DualRifles", 1)
		 Goto Ready

	Fire:
	TNT1 A 0
	TNT1 A 0 A_JumpIfInventory("FiredPrimary",1,"FireLeftGun") //Check if player wants to reload
	MPD2 A 1 //A_WeaponReady(WRF_NOFIRE)
	TNT1 A 0 A_JumpIfInventory("FiredPrimary",1,"FireLeftGun") //Check if player wants to reload
	TNT1 A 0 A_Refire
	Goto ReadyToFire

	ReadyToFire:
		TNT1 A 0
		TNT1 A 0 A_JumpIfInventory("FiredPrimary",1,"FireLeftGun") //Check if player wants to reload
		MPD2 A 1 A_WeaponReady// Weapon is ready to fire
		//TNT1 A 0 A_JumpIfInventory("IsFiringSecondary", 1, 1)
		//TNT1 A 0 A_GunFlash
		TNT1 A 0 A_JumpIfInventory("FiredPrimary",1,"FireLeftGun") //Check if player wants to reload
		TNT1 A 0 A_JumpIfInventory("Reloading",1,"Reload2") //Check if player wants to reload
		TNT1 A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		TNT1 A 0 A_JumpIfInventory("TossGrenade",1,"TossGrenade")
		TNT1 A 0 A_JumpIfInventory("GoWeaponSpecialAbility", 1, "WeaponSpecial")
        TNT1 A 0 A_JumpIfInventory("Taunting",1,"Taunt")
		TNT1 A 0 A_JumpIfInventory("Unloading",1,"Unload")
		Loop
		
	TossGrenade:
	    TNT1 A 0 A_TakeInventory("TossGrenade", 1)
	    TNT1 A 0 A_Print("You can't throw right now!")
	Goto ReadyToFire
		
	WeaponSpecial:
	TNT1 A 0 A_Takeinventory("GoWeaponSpecialAbility",1)
	TNT1 A 0 A_GiveInventory ("StopDualWield")
	TNT1 A 0 A_SelectWeapon ("MP40")
	Goto Ready
		
	Deselect:
		TNT1 AAAAAA 0 A_Lower
		TNT1 A 0 A_TakeInventory("TossGrenade", 1)
		TNT1 A 1 A_Lower //Makes player switch weapons faster
		Wait
	Select:
		TNT1 A 0 A_Raise //Makes the weapon always be selected fast
		Wait
	
	DryFire:
	    MPD2 A 0 A_PlaySound("DryFire/Rifle", 1)
		MPD2 A 10
		Goto ReadyToFire
	 
	FireLeftGun:
	    MPD2 A 0
		TNT1 A 0 A_JumpIfInventory("DoubleMP40Ammo", 1, 1)//First of all, checks if there is ammo inside the gun
		Goto Reload//Couldn't find any ammo. Reload
		
		TNT1 A 0 A_JumpIfInventory("LeftMP40Ammo", 1, 1)//Checks if there is ammo inside the left Rifle
		Goto DryFire//Couldn't find any ammo. Don't fire.
		
		TNT1 A 0 A_TakeInventory("DoubleMP40Ammo", 1)//Removes 1 ammo from magazine
		TNT1 A 0 A_AlertMonsters
		TNT1 A 0 A_FireCustomMissile("Tracer", random(-4,4), 0, -6, -2, 0, random(-2,2))
		MPD2 A 0 A_FireCustomMissile("GunFireSmoke", 0, 0, -5, 0, 0, 0)
		TNT1 A 0 A_Playsound ("MP40", Chan_Weapon)//Play gun fire sound
	    MPD2 B 1 BRIGHT A_FireBullets(12,3,-1,10,"MachineGunBulletPuff",1)//Horizontal Spread, Vertical Spread, Numbers of Projectiles (Leave as -1 if its a single one), Damage Factor
		TNT1 A 0 A_SetPitch(+0.6 + pitch)//Vertical recoil up
		MPD2 C 1 A_FireCustomMissile("RifleCaseSpawn",0,0,-20,-10) //Ejetts empty brass
		MPD2 C 0 A_SetPitch(-0.6 + pitch)//Vertical recoil down
		MPD2 D 1 A_TakeInventory("LeftMP40Ammo", 1)//Remove the fake ammo
		TNT1 A 0 A_JumpIfInventory("DoubleMP40Ammo", 1, 1)//Checks if there is ammo inside the gun
		Goto Reload//Couldn't find any ammo. Reload.
		MPD2 A 1
		TNT1 A 0 A_JumpIfInventory("LeftMP40Ammo", 1, 1)//Checks if there is ammo inside the left Rifle
		Goto DryFire//Couldn't find any ammo. Don't fire.
		TNT1 A 0 A_JumpIfInventory("FiredPrimary",1,"FireLeftGun") //Check if player still is firing left gun
		MPD2 A 0
		Goto Fire//No need to reload. Get back to Ready state.
		
	Flash:	
	    TNT1 A 0
		TNT1 A 0 A_JumpIfInventory("StopDualWield", 1, "LightDone")
		MPD1 A 1
		TNT1 A 0 A_JumpIfInventory("FiredSecondary", 1, "FireRightGun")
		//P2W1 A 1
		Loop
	
	DontFlash:
	    TNT1 A 1
		Goto Flash
	
	FireRightGun:
	    MPD1 A 0
		
		TNT1 A 0 A_JumpIfInventory("StopDualWield", 1, "LightDone")
		TNT1 A 0 A_GiveInventory("IsFiringSecondary", 1)
		TNT1 A 0 A_JumpIfInventory("DoubleMP40Ammo", 1, 2)//First of all, checks if there is ammo inside the gun
		TNT1 A 0 //A_GiveInventory("Reloading", 1)//Couldn't find any ammo. Reload
		Goto Flash//Couldn't find any ammo. Reload
		
		TNT1 A 0 A_JumpIfInventory("MP40Ammo", 1, 1)//Checks if there is ammo inside the left Rifle
		Goto DryFire2//Couldn't find any ammo. Don't fire.
		
		TNT1 A 0 A_TakeInventory("DoubleMP40Ammo", 1)//Removes 1 ammo from magazine
		
		TNT1 A 0 A_AlertMonsters
		TNT1 A 0 A_FireCustomMissile("Tracer", random(-4,4), 0, 6, -2, 0, random(-2,2))
		RIFF A 0 A_FireCustomMissile("GunFireSmoke", 0, 0, 5, 0, 0, 0)
		TNT1 A 0 A_Playsound ("MP40", Chan_Weapon)//Play gun fire sound
	    MPD1 B 1 BRIGHT A_FireBullets(12,3,-1,10,"MachineGunBulletPuff",1)//Horizontal Spread, Vertical Spread, Numbers of Projectiles (Leave as -1 if its a single one), Damage Factor
		TNT1 A 0 A_SetPitch(+0.6 + pitch)//Vertical recoil up
		MPD1 C 1 A_FireCustomMissile("RifleCaseSpawn",0,0,20,-10) //Ejetts empty brass
		MPD1 C 0 A_SetPitch(-0.6 + pitch)//Vertical recoil down
		MPD1 D 1 A_TakeInventory("MP40Ammo", 1)//Remove the fake ammo
		MPD1 A 0
		TNT1 A 0 A_JumpIfInventory("DoubleMP40Ammo", 1, 3)//Checks if there is ammo inside the gun
		TNT1 A 0 //A_GiveInventory("Reloading", 1)//Couldn't find any ammo. Reload
		TNT1 A 0 A_JumpIfInventory("MP40Ammo", 1, 1)//Checks if there is ammo inside the right Rifle
		Goto DryFire2//Couldn't find any ammo. Don't fire.
		MPD1 A 1
		TNT1 A 0 A_JumpIfInventory("FiredSecondary", 1, "FireRightGun")
		Goto Flash
		
	DryFIre2:
	    MPD1 A 1 A_PlaySound("DryFire/Rifle", 1)
		MPD1 AAAAAAAAAA 1 A_JumpIfInventory("StopDualWield", 1, "LightDone")
		Goto Flash
		
		
		
		
		
		
		
		
		
	Reload:
	    
	    MPDG A 1 //Always leave one free line before the reload animation to prevent bugs
		TNT1 A 0 A_TakeInventory("Reloading")//Always put this to make sure that the game understand that you already are reloading your weapon, and calling this state again is not necessary
		TNT1 A 0 A_JumpIfInventory("DoubleMP40Ammo", 32, "ReadytoFire")//Checks if the weapon is not fully reloaded, if yes, cancel reload.
		TNT1 A 0 A_PlaySound("DryFire/Rifle", 1)
		TNT1 A 0 A_JumpIfInventory("P.Bullets", 1, 1)//Checks if you have spare ammo
		Goto DryFire//Identified that you have no spare ammo. Get back to Ready animation
		TNT1 A 0 A_GiveInventory("StopDualWield", 1)
        
	Refill:	
	    TNT1 A 0
		TNT1 A 0 A_JumpIfInventory("P.Bullets", 1, 1)//Check if you have spare ammo again. This line is necessary in case you don't have enough ammo to fill an entire clip.
		Goto FinishReload//Gun is fully loaded. Go to finish animation
	    TNT1 A 0 A_TakeInventorY("P.Bullets", 1)//Removes one spare bullet
		TNT1 A 0 A_GiveInventory("DoubleMP40Ammo", 1)//Puts one bullet inside the clip
		TNT1 A 0 A_JumpIfInventory("DoubleMP40Ammo", 32, "FinishReload")//Checks if weapons is full
		Loop
		
	 FinishReload:	//Weapon is fully loaded. Time to play the reloading animation.
		TNT1 A 0
		TNT1 A 0 A_JumpIfInventory("LeftMP40Ammo", 32, "FinishReloadRightOnly")
		TNT1 A 0 A_JumpIfInventory("MP40Ammo", 32, "FinishReloadLeftOnly")
		TNT1 A 0 A_GiveInventory("LeftMP40Ammo", 32)
		TNT1 A 0 A_GiveInventory("MP40Ammo", 32)
		MPDS FEDCBA 1
		TNT1 A 5

		MP4R B 1 A_PlaySound("SGCOUT")
		MP4R C 2
		MP4R DE 1
		TNT1 A 0 A_JumpIfInventory("HasUnloaded", 1, 2)
		MP4R A 0 A_CustomMissile("Mp40DeathClip", 7, 0, random(-60,70), 2, random(20,40))
		MP4R F 1
		MP4R G 12
        MP4R GFEDCB 1
		MP4R A 1 A_PlaySound("SGCIN")
		MP4R A 2
		MP4G A 1
		TNT1 A 0 A_PlaySound ("MP_Bolt", Chan_Weapon)
		
		TNT1 A 10
		
		MP2R B 1 A_PlaySound("SGCOUT")
		MP2R C 2
		MP2R DE 1
		TNT1 A 0 A_JumpIfInventory("HasUnloaded", 1, 2)
		MP4R A 0 A_CustomMissile("Mp40DeathClip", 7, 0, random(-60,70), 2, random(20,40))
		MP2R F 1
		MP2R G 12
        MP2R GFEDCB 1
		MP4R A 1 A_PlaySound("SGCIN")
		MP2R A 2
		MP2R A 1
		TNT1 A 0 A_PlaySound ("MP_Bolt", Chan_Weapon)
		TNT1 A 0 A_TakeInventory("StopDualWield", 1)
		TNT1 A 0 A_TakeInventory("FiredPrimary", 1)
		TNT1 A 0 A_TakeInventory("FiredSecondary", 1)
		TNT1 A 5
		MPDS ABCDEF 1
		TNT1 A 0 A_TakeInventory("Reloading")//Always put this to make sure that the game understand that you already are reloading your weapon, and calling this state again is not necessary
		TNT1 A 0 A_GunFlash
		TNT1 A 0 A_Takeinventory("HasUnloaded",1)
		Goto ReadyToFire
		
		
		
		
		//Reload 2 is for manual reload, leaves one more bullet inside the gun.
		
	Reload2:
	    MPDG A 1 //Always leave one free line before the reload animation to prevent bugs
		TNT1 A 0 A_TakeInventory("Reloading")//Always put this to make sure that the game understand that you already are reloading your weapon, and calling this state again is not necessary
		TNT1 A 0 A_JumpIfInventory("DoubleMP40Ammo", 32, "ReadytoFire")//Checks if the weapon is not fully reloaded, if yes, cancel reload.
		TNT1 A 0 A_JumpIfInventory("P.Bullets", 1, 1)//Checks if you have spare ammo
		Goto ReadyToFire//The game identified that you have no spare ammo. Get back to Ready animation
		TNT1 A 0 A_GiveInventory("StopDualWield", 1)
        
	Refill2:	
	    TNT1 A 0
		TNT1 A 0 A_JumpIfInventory("P.Bullets", 1, 1)//Check if you have spare ammo again. This line is necessary in case you don't have enough ammo to fill an entire clip.
		Goto FinishReload//Gun is fully loaded. Go to finish animation
	    TNT1 A 0 A_TakeInventorY("P.Bullets", 1)//Removes one spare bullet
		TNT1 A 0 A_GiveInventory("DoubleMP40Ammo", 1)//Puts one bullet inside the clip
		TNT1 A 0 A_JumpIfInventory("DoubleMP40Ammo", 32, "FinishReload2")//Checks if weapons is full
		Loop
		
		
	FinishReload2:	//Weapon is fully loaded. Time to play the reloading animation.
	    TNT1 A 0
		TNT1 A 0 A_JumpIfInventory("LeftMP40Ammo", 32, "FinishReloadRightOnly")
		TNT1 A 0 A_JumpIfInventory("P.Bullets", 32, "FinishReloadLeftOnly")
		TNT1 A 0 A_GiveInventory("LeftMP40Ammo", 32)
		TNT1 A 0 A_GiveInventory("P.Bullets", 32)
		MPDS FEDCBA 1
		TNT1 A 5
		MP4R B 1 A_PlaySound("SGCOUT")
		MP4R C 2
		MP4R DE 1
		TNT1 A 0 A_JumpIfInventory("HasUnloaded", 1, 2)
		MP4R A 0 A_CustomMissile("Mp40DeathClip", 7, 0, random(-60,70), 2, random(20,40))
		MP4R F 1
		MP4R G 12
        MP4R GFEDCB 1
		MP4R A 1 A_PlaySound("SGCIN")
		MP4R A 2
		MP4G A 1
		TNT1 A 0 A_PlaySound ("MP_Bolt", Chan_Weapon)
		
		TNT1 A 10
		
		MP2R B 1 A_PlaySound("SGCOUT")
		MP2R C 2
		MP2R DE 1
		TNT1 A 0 A_JumpIfInventory("HasUnloaded", 1, 2)
		MP4R A 0 A_CustomMissile("Mp40DeathClip", 7, 0, random(-60,70), 2, random(20,40))
		MP2R F 1
		MP2R G 12
        MP2R GFEDCB 1
		MP4R A 1 A_PlaySound("SGCIN")
		MP2R A 2
		MP2R A 1
		TNT1 A 0 A_PlaySound ("MP_Bolt", Chan_Weapon)
		TNT1 A 0 A_TakeInventory("StopDualWield", 1)
		TNT1 A 0 A_TakeInventory("FiredPrimary", 1)
		TNT1 A 0 A_TakeInventory("FiredSecondary", 1)
		TNT1 A 5
		MPDS ABCDEF 1
		TNT1 A 0 A_TakeInventory("Reloading")//Always put this to make sure that the game understand that you already are reloading your weapon, and calling this state again is not necessary
		TNT1 A 0 A_GunFlash
		TNT1 A 0 A_Takeinventory("HasUnloaded",1)
		Goto ReadyToFire	
		
		
	
	FinishReloadLeftOnly:	//Weapon is fully loaded. Time to play the reloading animation.
		TNT1 A 0 A_GiveInventory("LeftMP40Ammo", 32)
		TNT1 A 0 A_GiveInventory("P.Bullets", 32)
		MPDS FEDCBA 1
		MP2R B 1 A_PlaySound("SGCOUT")
		MP2R C 2
		MP2R DE 1
		TNT1 A 0 A_JumpIfInventory("HasUnloaded", 1, 2)
		MP4R A 0 A_CustomMissile("Mp40DeathClip", 7, 0, random(-60,70), 2, random(20,40))
		MP2R F 1
		MP2R G 12
        MP2R GFEDCB 1
		MP2R A 1 A_PlaySound("SGCIN")
		MP2R A 2
		MP2R A 1
		TNT1 A 0 A_PlaySound ("MP_Bolt", Chan_Weapon)
		TNT1 A 0 A_TakeInventory("StopDualWield", 1)
		TNT1 A 0 A_TakeInventory("FiredPrimary", 1)
		TNT1 A 0 A_TakeInventory("FiredSecondary", 1)
		TNT1 A 5
		MPDS ABCDEF 1
		TNT1 A 0 A_TakeInventory("Reloading")//Always put this to make sure that the game understand that you already are reloading your weapon, and calling this state again is not necessary
		TNT1 A 0 A_GunFlash
		TNT1 A 0 A_Takeinventory("HasUnloaded",1)
		Goto ReadyToFire	
		
	FinishReloadRightOnly:	//Weapon is fully loaded. Time to play the reloading animation.
		TNT1 A 0 A_GiveInventory("LeftMP40Ammo", 32)
		TNT1 A 0 A_GiveInventory("P.Bullets", 32)
		MPDS FEDCBA 1
		MP4R B 1 A_PlaySound("SGCOUT")
		MP4R C 2
		MP4R DE 1
		TNT1 A 0 A_JumpIfInventory("HasUnloaded", 1, 2)
		MP4R A 0 A_CustomMissile("Mp40DeathClip", 7, 0, random(-60,70), 2, random(20,40))
		MP4R F 1
		MP4R G 12
        MP4R GFEDCB 1
		MP4R A 1 A_PlaySound("SGCIN")
		MP4R A 2
		MP4R A 1
		TNT1 A 0 A_PlaySound ("MP_Bolt", Chan_Weapon)
		TNT1 A 0 A_TakeInventory("StopDualWield", 1)
		TNT1 A 0 A_TakeInventory("FiredPrimary", 1)
		TNT1 A 0 A_TakeInventory("FiredSecondary", 1)
		TNT1 A 5
		MPDS ABCDEF 1
		TNT1 A 0 A_TakeInventory("Reloading")//Always put this to make sure that the game understand that you already are reloading your weapon, and calling this state again is not necessary
		TNT1 A 0 A_GunFlash
		TNT1 A 0 A_Takeinventory("HasUnloaded",1)
		Goto ReadyToFire		
		
		
		
	 DoKick:
		NULL A 0 A_JumpIf (momZ > 0, "AirKick")
		NULL A 0 A_JumpIf (momZ < 0, "AirKick")
		
        TNT1 A 0 A_jumpifinventory("PowerStrength",1,"BerserkerKick")
		TNT1 A 0 A_PlaySound("KICK")
		TNT1 A 0 SetPlayerProperty(0,1,0)
		J2DK ABC 1
		RIFF A 0 A_FireCustomMissile("KickAttack", 0, 0, 0, -7)
        J2DK D 6
		J2DK EECBA 1
		KICK A 0 A_Takeinventory("Kicking",1)
		TNT1 A 0 SetPlayerProperty(0,0,0)
		TNT1 A 0 A_JumpIfInventory("GoFatality", 1, "Steady")
		Goto ReadyToFire
    BerserkerKick:
		TNT1 A 0 A_PlaySound("KICK")
		TNT1 A 0 SetPlayerProperty(0,1,0)
		J2DK ABC 1
		RIFF A 0 A_FireCustomMissile("SuperKickAttack", 0, 0, 0, -7)
        J2DK D 6
		J2DK EECBA 1
		TNT1 A 0 SetPlayerProperty(0,0,0)
		TNT1 A 0 A_JumpIfInventory("GoFatality", 1, "Steady")
		Goto ReadyToFire
	
	AirKick:
	    TNT1 A 0 A_jumpifinventory("PowerStrength",1,"SuperAirKick")
		TNT1 A 0 A_PlaySound("KICK")
		TNT1 A 0 A_Recoil (-6)
		KLCK JKLMN 1
        RIFF A 0 A_FireCustomMissile("AirKickAttack", 0, 0, 0, -31)
        KLCK O 3
		KLCK A 0 A_Takeinventory("Kicking",1)
		KLCK PQRST 2
		TNT1 A 0 A_JumpIfInventory("GoFatality", 1, "Steady")
		Goto ReadyToFire
		
	SuperAirKick:
		TNT1 A 0 A_PlaySound("KICK")
		TNT1 A 0 A_Recoil (-6)
		KLCK JKLMN 1
        RIFF A 0 A_FireCustomMissile("SuperAirKickAttack", 0, 0, 0, -31)
        KLCK O 3
		KLCK A 0 A_Takeinventory("Kicking",1)
		KLCK PQRST 2
		TNT1 A 0 A_JumpIfInventory("GoFatality", 1, "Steady")
		Goto ReadyToFire
	Taunt:
		TNT1 A 0 A_Takeinventory("Zoomed",1)
        TNT1 A 0 A_ZoomFactor(1.0)
        TNT1 A 10
		FUCK F 1
		TNT1 A 0 BRIGHT A_FireCustomMissile("Taunter", 0, 0, -1, 0)
		TNT1 A 0 BRIGHT A_FireCustomMissile("Taunter", -9, 0, -1, 0)
		TNT1 A 0 BRIGHT A_FireCustomMissile("Taunter", 9, 0, -1, 0)
        FUCK G 1 A_PlaySound("FUCK", 2)
        FUCK HI 1 A_AlertMonsters
		FUCK J 15 A_Takeinventory("Taunting",1)
        FUCK IHGF 1
        TNT1 A 10
		Goto ReadyToFire

	Steady:
	TNT1 A 1
	Goto ReadyToFire	
		
		
	Unload:
		TNT1 A 1 A_WeaponReady(WRF_NOBOB)
		TNT1 A 0 A_ZoomFactor(1.0)
		TNT1 A 0 A_Takeinventory("Unloading",1)
		TNT1 A 0 A_Takeinventory("ADSmode",1)
		TNT1 A 0 A_Takeinventory("Zoomed",1)
        TNT1 A 0 A_JumpIfInventory("DoubleMP40Ammo",1,3)
        Goto Ready
        
		
	RemoveBullets:
		TNT1 AAAA 0
		TNT1 A 0 A_JumpIfInventory("DoubleMP40Ammo",1,3)
		Goto FinishUnload
        TNT1 AAAAAA 0
		TNT1 A 0 A_Takeinventory("MP40Ammo",1)
		TNT1 A 0 A_Takeinventory("DoubleMP40Ammo",1)
		TNT1 A 0 A_Takeinventory("LeftMP40Ammo",1)
		TNT1 A 0 A_Giveinventory("NaziAmmo",1)
		Goto RemoveBullets
	
	FInishUnload:
		TNT1 AAA 0
		TNT1 A 0 A_GiveInventory("StopDualWield")
		TNT1 A 0 A_Takeinventory("Zoomed",1)
        TNT1 A 0 A_PlaySound("Reload")
		TNT1 A 0 A_GiveInventory ("Pumping", 1)
		TNT1 A 0 A_Takeinventory("Unloading",1)
		R2RL ONMLKJIHGFEDCBA 1
		R2RR ONMLKJIHGFEDCBA 1
		TNT1 A 0 A_GiveInventory("HasUnloaded", 1)
		TNT1 A 0 A_TakeInventory("StopDualWield")
		TNT1 A 0 A_Takeinventory("Unloading",1)
		Goto Ready+6
		
	}
}

Actor MP40PickUp : CustomInventory 29307
{
  +THRUACTORS +FORCEXYBILLBOARD
  Inventory.PickupMessage "You Found the Nazi MP40! (Slot 4)"
  Inventory.Pickupsound "RIFLE"
  Scale 0.5
  //$Category Weapons/Dual Weapons
  States
  {
  Spawn:
    MP40 E 1
	TNT1 A 0 A_ChangeFlag("THRUACTORS", 0)
    loop
	
  PickUp:
   TNT1 A 0
   TNT1 A 0 A_JumpIfInventory("MP40", 1, "DualRifle")
   TNT1 A 0 A_JumpIfInventory("MP-86s", 1, "UpgradeMP40")
